CREATE TABLE HodnoceniRidice (
    IDHodnoceniRidice int NOT NULL PRIMARY KEY,
    Obsah varchar(255),
    PocetHvezdicek int
);

CREATE TABLE HodnoceniSpolucestujiciho (
    IDHodnoceniSpolucestujiciho int NOT NULL PRIMARY KEY,
    Obsah varchar(255),
    Spokojenost int,
    Dochvilnost int,
    Pratelstvi int
);


CREATE TABLE Uzivatel (
    IDUzivatel int NOT NULL PRIMARY KEY,
    Email varchar(255),
    Telefon varchar(255),
    Popis varchar(255),
    CestaKSouboruFotografie varchar(255),
    Hudba number(1,0),
    Kouření number(1,0), 
    Zvířata number(1,0), 
    Komunikace varchar(255),
    Zkušenost int
);

CREATE TABLE Auto (
    RegistracniZnacka varchar(255) NOT NULL PRIMARY KEY,
    Znacka varchar (255),
    Model varchar (255),
    Kapacita int
);

CREATE TABLE Spolujizda (
    IDSpolujizda int NOT NULL PRIMARY KEY,
    CasVyjezdu varchar(255),
    NastupniMisto varchar(255),
    VystupniMisto varchar(255),
    Cena int,
    Zajizdka varchar(255),
    CasovaFlexibilita varchar(255),
    Zavazadla number(1,0)
);

CREATE TABLE Clanek (
    IDClanek int NOT NULL PRIMARY KEY,
    Obsah varchar(255)
);

CREATE TABLE Opravneni (
    IDOpravneni int NOT NULL PRIMARY KEY,
    Typ int
);

CREATE TABLE Vlog (
    IDVlog int NOT NULL PRIMARY KEY,
    Popis varchar(255),
    CestaKSouboruVidea varchar(255)
);

CREATE TABLE Vylet (
    IDVylet int NOT NULL PRIMARY KEY,
    PopisProgramu varchar(255),
    Ubytovani varchar(255),
    Naklady int,
    Narocnost int
);

CREATE TABLE Aktivita (
    IDAktivita int NOT NULL PRIMARY KEY,
    Nazev varchar(255),
    Druh varchar(255)
);

CREATE TABLE Vybaveni (
    IDVybaveni int NOT NULL PRIMARY KEY,
    Nazev varchar(255),
    Druh varchar(255)
);

CREATE TABLE Misto (
    IDMisto int NOT NULL PRIMARY KEY,
    Nazev varchar(255)
);

CREATE TABLE Stat (
    IDKodStatu varchar(255) NOT NULL PRIMARY KEY,
    Nazev varchar(255)  
);
